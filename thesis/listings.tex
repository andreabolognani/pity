	{morekeywords={lambda,
                   define, define-struct, let, let*,
                   list, display},
     sensitive=true,
     morecomment=[l]{;},
     morestring=[b]{"},}

        basicstyle=\ttfamily\small,
        keywordstyle=\bf,
        emphstyle=\underbar,
        showstringspaces=false,
        numbers=left,
        numberstyle=\tiny,
        numbersep=10pt}
